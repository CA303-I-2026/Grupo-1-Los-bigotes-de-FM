%==============================
\newcommand{\prefaciostyle}{
	\titleformat{\chapter}[display]
	{\huge\bfseries}  % Estilo sin número grande
	{}  % Sin "Capítulo X"
	{0pt}  % Sin espacio entre número y título
	{}  % No mostrar número de capítulo
	% --- Ajustar el espacio antes y después del capítulo ---
	\titlespacing*{\chapter}{0pt}{-20pt}{260pt}
}

%==============================
% ESTILO PARA EL RESTO DE LOS CAPÍTULOS
%==============================
\newcommand{\lchaptersstyle}{
	\titleformat{\chapter}[display]
	{\Huge\bfseries}  % Estilo del título (grande y en negrita)
	{}  % Sin "Capítulo X"
	{-150pt}  % Espacio entre número y título
	{\fontsize{50}{60}\selectfont\thechapter \Huge\\}  % Mostrar número grande para capítulos
	% --- Ajustar el espacio antes y después del capítulo ---
	\titlespacing*{\chapter}{0pt}{100pt}{260pt}
}

%==============================
% ESTILO PARA CAPÍTULOS DERECHOS
%==============================
\newcommand{\rchaptersstyle}{
	\titleformat{\chapter}[display]
	{\Huge\bfseries\raggedleft}  % Estilo + alineación derecha
	{}  % Sin "Capítulo X"
	{-150pt}  % Espacio entre número y título
	{\fontsize{50}{60}\selectfont\raggedleft\thechapter \Huge\\}  % Número alineado a la derecha
	\titlespacing*{\chapter}{0pt}{100pt}{260pt}  % Espaciado antes y después
}
